Element-plus的el-tree-select组件默认选中及显示

Posted at — Nov 08, 2022

如官方描述，el-tree-select是含有下拉菜单的树形选择器，结合了 el-tree 和 el-select 两个组件的功能。这个控件用来处理树形结构的选择十分方便，最近我在做后台角色组权限选择的时候用到它。

我要实现的是这样的一个功能界面：

在新增的场景下没有出现问题，在编辑的场景下，由于需要给 el-tree-select指定默认选中的值，其中用到default-checked-keys这个属性，不过在使用过程中出现了一些小问题，值得记录下来。

<template>
 <el-form :model="form" label-width="120px" :rules="formRules" ref="form">
      <el-form-item label="组别名称" prop="name">
        <el-input v-model="form.name" placeholder="请输入组别名称" />
      </el-form-item>

      <el-form-item label="权限" prop="auths">
        <el-tree-select
            :props="defaultProps"
            v-model="form.auths"
            :data="tree"
            node-label="title"
            node-key="id"
            :render-after-expand="false"
            :default-checked-keys="form.auths"
            show-checkbox
            check-strictly
            check-on-click-node
            multiple
            clearable
            filterable
        />
      </el-form-item>
      <el-form-item label="状态">
      <!--默认情况下，el-switch需绑定true/false,指定active-value可以使用数字，这里由于后台返回的状态是0/1，需要指定一下这个属性,用active-value绑定要启用状态的值,用inactive-value绑定禁用状态的值-->
      <el-switch v-model="form.status" :active-value="1" :inactive-value="0"/>
      </el-form-item>
      <el-form-item>
      <el-button type="primary" @click="onSubmit">保存</el-button>
      <el-button @click="closeDialog">取消</el-button>
      </el-form-item>
  </el-form>
  </template>

<script>
import { getMenuList,saveGroup } from "@/api/api";

export default {
  name: 'AddGroup',
  props:{
        item: {
            type:Object
        }
  },
  watch: {
    //监听待编辑的对象，由父页面传过来。
    /*
    item对象结构如下：
    {id:1,name:'管理员',auths:'1,2,3',status:1}
    */
    item: {
        deep: true,
        immediate:true,
        handler(newValue, oldValue) {
            if (newValue) {
                this.form = newValue
                if (typeof(newValue.auths)=='string') {
                	  //由于我们要记住原有的数据，需给el-tree-select指定默认选中项，指定default-checked-keys这个属性即可，这里传入选中的id数组。
                	  //将后台返回的auths字段的字符串转为数组，重新给form的auths赋值
                    const list = newValue.auths.split(",")
                    this.form.auths = list
                }
            } else {
                this.form = {
                    auths:[],
                    status:1
                }
            }
        }
    },
  },
  data() {
      return {
        formRules: {
            name:[{required:true,message:'组别名称不能为空',trigger:'blur'}],
            auths:[{required:true,message:'请选择权限',trigger:'change'}],
        },
        form:{
            id:'',
            name:'',
            auths:[],
            status:1
        },
        //配置el-tree-select的自定义label,value字段名称
        defaultProps: {
            children: 'children',
            label: 'title',
            value: 'id',
        },
        tree:[],
      }
  },
  mounted() {
    this.getListApi()
  },
  methods: {
  	  //获取菜单列表
      getListApi() {
          var that = this
          const params = {}
          getMenuList(params)
          .then((res) => {
              if (res.code == 1) {
                that.tree = res.data.data
              }
          })
          .catch((err) => console.log(err));            
      },

  }
}
</script>
